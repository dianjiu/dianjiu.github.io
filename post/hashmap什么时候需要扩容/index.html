HashMap什么时候需要扩容？怎么实现的？

2021-08-14
面试笔记

文章目录

    • 扩容概述
    • 扩容场景
    • 实现过程

扩容概述

扩容需要分配一个新的数组，新数组的长度是老数组长度的两倍，然后遍历整个老数组，把其中的每个元素全部重新Hash计算，分配到新的数组结构中。

扩容场景

  • 初始化数组Table的时候进行扩容
  • 当数组table的size达到阀值的时候进行扩容

实现过程

​ 通过判断旧数组的容量是否大于0来判断数组是否初始化过。

  • 如果小于0：进行初始化,判断是否调用无参构造器。

    如果调用了无参构造器: 使用默认的大小和阙值<阈值 12（阀值=默认大小为16乘以负载因子0.75）。

    //HashMap的默认大小为16（1向左移动4位）
static final int DEFAULT_INITIAL_CAPACITY = 1 << 4;
//最大容量
static final int MAXIMUM_CAPACITY = 1 << 30;
//默认负载因子
static final float DEFAULT_LOAD_FACTOR = 0.75f;
//负载因子
final float loadFactor;
//扩容因子(阀值)=容量*负载因子
int threshold;
//午饭构造方法
public HashMap() {
        //无参构造方法中，负载因子默认为0.75
        this.loadFactor = DEFAULT_LOAD_FACTOR; 
    }

    如果没有调用无参构造器: 使用构造函数中初始化的容量, 当然这个容量是经过tableSizefor 计算后的 2 的次幂数) 。

    //全参构造方法
public HashMap(Map<? extends K, ? extends V> m) {
        //初始化负载因子为0.75
        this.loadFactor = DEFAULT_LOAD_FACTOR;
        //放置集合新元素
        putMapEntries(m, false);
    }
//添加元素
final void putMapEntries(Map<? extends K, ? extends V> m, boolean evict) {
        //获取map集合的大小
        int s = m.size();
        //通过判断旧数组的容量是否大于0来判断数组是否初始化过
        if (s > 0) {
            //初始化数组table
            if (table == null) { // pre-size
                //由现有大小除以负载系数(默认0.75)后再加1，得到新集合容量ft
                float ft = ((float)s / loadFactor) + 1.0F;
                //如果新集合容量大小小于最大值，新集合容量大小为ft，否则为最大值 1>>30
                int t = ((ft < (float)MAXIMUM_CAPACITY) ?
                         (int)ft : MAXIMUM_CAPACITY);
                //长度大于扩容因子(阀值)时，新集合的大小为ft
                if (t > threshold)
                    //经过tableSizefor 计算后的 2 的次幂数
                    threshold = tableSizeFor(t);
            } 
            //此处为JDK13的源码
            else {
                //JDK13中更加严谨
                //长度大于扩容因子(阀值)并且长度小于最大容量时，进行扩容
                while (s > threshold && table.length < MAXIMUM_CAPACITY)
                    //扩容方法的具体实现
                    resize();
            }
            //此处注释掉的代码为JDK8的源码
            //长度大于扩容因子(阀值)时，进行扩容
            //else if (s > threshold)
                //扩容方法的具体实现
               // resize();
            //遍历整个老数组
            for (Map.Entry<? extends K, ? extends V> e : m.entrySet()) {
                K key = e.getKey();
                V value = e.getValue();
                //从新计算hash，放到新的数组中
                putVal(hash(key), key, value, false, evict);
            }
        }
    }

  • 如果大于0: 进行扩容，扩容成两倍(小于最大值的情况下)，之后在进行将元素重新进行与运算复制到新的散列表中。

//基于zulu JDK13 的扩容源码
final Node<K,V>[] resize() {
       //原始table
        Node<K,V>[] oldTab = table;
       //老的容量
        int oldCap = (oldTab == null) ? 0 : oldTab.length;
       //老的阀值
        int oldThr = threshold;
        //初始化新的table容量、阀值为0
        int newCap, newThr = 0;
        //判断原始table的容量是否大于0
        if (oldCap > 0) {
            //判断原始table的容量是否大于最大容量
            if (oldCap >= MAXIMUM_CAPACITY) {
                //如果原始table的容量大于等于最大值了，设置阀值为最大容量值
                threshold = Integer.MAX_VALUE;
                //然后不做任何处理，直接返回原始table对象
                return oldTab;
            }
            //如果原始table的大小小于最大容量，则把新的table的容量设置为原始table的容量的2倍（1oldCap<<1）
            //同时判断新的table的容量是否小于最大值，并且判断原始的table的容量是否大于等于初始化容量16
            else if ((newCap = oldCap << 1) < MAXIMUM_CAPACITY &&
                     oldCap >= DEFAULT_INITIAL_CAPACITY)
                //满足条件的话，新的table的阀值也扩容为原始table阀值的2倍
                newThr = oldThr << 1; // double threshold
        }
       //当原始table的容量小于或等于0，但是原始table的阀值大于0
       //扩容一个指定容量为原始table阀值的新table
        else if (oldThr > 0) // initial capacity was placed in threshold
            //新的table的容量为原始table的阀值，即为12
            newCap = oldThr;
       // 这里是第一次初始化一个新的table
        else {               // zero initial threshold signifies using defaults
            //新的table的容量为16
            newCap = DEFAULT_INITIAL_CAPACITY;
            //新的table的阀值为12
            newThr = (int)(DEFAULT_LOAD_FACTOR * DEFAULT_INITIAL_CAPACITY);
        }
        //如果新table的阀值为0，则重置新table的阀值为新table容量*负载因子
        if (newThr == 0) {
            float ft = (float)newCap * loadFactor;
            newThr = (newCap < MAXIMUM_CAPACITY && ft < (float)MAXIMUM_CAPACITY ?
                      (int)ft : Integer.MAX_VALUE);
        }
        //设置默认的负载因子为新的table的负载因子
        threshold = newThr;
        //告诉编译器同时忽略rawtypes和unchecked警告信息。
        @SuppressWarnings({"rawtypes","unchecked"})
        //初始化一个新的table对象，指定容量
        Node<K,V>[] newTab = (Node<K,V>[])new Node[newCap];
        //设置默认table对象为新的table
        table = newTab;
       //当原始table不为空的时候，遍历原始table，并添加到新的table的上
        if (oldTab != null) {
            for (int j = 0; j < oldCap; ++j) {
                Node<K,V> e;
                if ((e = oldTab[j]) != null) {
                    oldTab[j] = null;
                    if (e.next == null)
                        newTab[e.hash & (newCap - 1)] = e;
                    else if (e instanceof TreeNode)
                        ((TreeNode<K,V>)e).split(this, newTab, j, oldCap);
                    else { // preserve order
                        Node<K,V> loHead = null, loTail = null;
                        Node<K,V> hiHead = null, hiTail = null;
                        Node<K,V> next;
                        do {
                            next = e.next;
                            if ((e.hash & oldCap) == 0) {
                                if (loTail == null)
                                    loHead = e;
                                else
                                    loTail.next = e;
                                loTail = e;
                            }
                            else {
                                if (hiTail == null)
                                    hiHead = e;
                                else
                                    hiTail.next = e;
                                hiTail = e;
                            }
                        } while ((e = next) != null);
                        if (loTail != null) {
                            loTail.next = null;
                            newTab[j] = loHead;
                        }
                        if (hiTail != null) {
                            hiTail.next = null;
                            newTab[j + oldCap] = hiHead;
                        }
                    }
                }
            }
        }
        // 扩容后，返回新的table对象
        return newTab;
    }
